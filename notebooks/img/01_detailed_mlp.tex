% varaibles
\providecommand{\inText}[1]{
 $x_{#1}$
}
\providecommand{\hidlText}[1]{
 {$H_{#1} = \sum_{j} \mathbf{\omega}^{I}_{#1 j} \cdot x_{j} + \beta^{I}_{#1}$}%
}
\providecommand{\hidrText}[1]{
 {$h_{#1}=\sigma_{relu}(H_{#1})$}%
}
\providecommand{\outlText}[1]{
 {$y'_{#1} = \sum_{j} \mathbf{\omega}^{II}_{#1 j} \cdot h_{j} + \beta^{II}_{#1}$}%
}

\providecommand{\outrText}[1]{
 $\hat{y}_{#1}=\sigma(y'_{#1})$%
}
\providecommand{\synapseIHText}[2]{%
  {\ifthenelse{\equal{#2}{0}}{\small$\boldsymbol{\omega}^{(I)}_{#1#2}$}{\dots}}}
\providecommand{\synapseHOText}[2]{%
  {\ifthenelse{\equal{#2}{0}}{\small$\boldsymbol{\omega}^{(II)}_{#1#2}$}{\dots}}%
}
